Spring Initializr

Générateur de projets Spring Boot sur start.spring.io — votre point de départ pour chaque microservice.

Spring Initializr crée automatiquement la structure de base d'un projet avec toutes les dépendances configurées.

Analogie : C'est comme un architecte qui dessine les plans de votre maison. Vous décrivez vos besoins (nombre de pièces, style), et il produit des plans conformes aux normes de construction.

Pourquoi utiliser Spring Initializr ?

  • Génère une structure standardisée
  • Configure automatiquement Maven/Gradle
  • Sélectionne les versions compatibles entre elles
  • Évite les erreurs de configuration initiale

Accessible sur start.spring.io, il est également disponible directement dans IntelliJ IDEA via File → New → Project → Spring Initializr.

5.1.7 Service JWT

javacom/ecommerce/authservice/service/JwtService.java
package com.ecommerce.authservice.service;

import io.jsonwebtoken.Claims;
import io.jsonwebtoken.Jwts;
import io.jsonwebtoken.security.Keys;
import jakarta.annotation.PostConstruct;
import java.nio.charset.StandardCharsets;
import java.time.Instant;
import java.time.temporal.ChronoUnit;
import java.util.Date;
import java.util.Set;
import javax.crypto.SecretKey;
import lombok.extern.slf4j.Slf4j;
import org.springframework.beans.factory.annotation.Value;
import org.springframework.stereotype.Service;

/**
 * @Slf4j injecte le logger automatiquement.
 * @RequiredArgsConstructor non applicable ici : le constructeur utilise @Value
 * pour injecter des propriétés de configuration — Spring ne peut pas résoudre
 * ces annotations via un constructeur généré par Lombok.
 */
@Slf4j
@Service
public class JwtService {

    private final String secret;
    private final long expirationMinutes;
    private SecretKey key;

    public JwtService(
            @Value("${jwt.secret}") String secret,
            @Value("${jwt.expiration-minutes:60}") long expirationMinutes) {
        this.secret = secret;
        this.expirationMinutes = expirationMinutes;
    }

    @PostConstruct
    void init() {
        this.key = Keys.hmacShaKeyFor(secret.getBytes(StandardCharsets.UTF_8));
    }
    
    public String generateToken(String email, Set<String> roles) {
        Instant now = Instant.now();
        
        return Jwts.builder()
                .subject(email)
                .claim("roles", roles)
                .issuedAt(Date.from(now))
                .expiration(Date.from(now.plus(expirationMinutes, ChronoUnit.MINUTES)))
                .signWith(key)
                .compact();
    }
    
    public Claims validateToken(String token) {
        return Jwts.parser()
                .verifyWith(key)
                .build()
                .parseSignedClaims(token)
                .getPayload();
    }
}

Sécurité : Le secret JWT doit faire au minimum 256 bits (32 caractères). En production, stockez-le dans un gestionnaire de secrets (Vault, GCP Secret Manager).

5.1.9 Implémentation UserServiceImpl

javacom/ecommerce/authservice/service/impl/UserServiceImpl.java
package com.ecommerce.authservice.service.impl;

import com.ecommerce.authservice.domain.dto.AuthRequest;
import com.ecommerce.authservice.domain.dto.AuthResponse;
import com.ecommerce.authservice.domain.dto.RegisterRequest;
import com.ecommerce.authservice.domain.entity.Role;
import com.ecommerce.authservice.domain.entity.User;
import com.ecommerce.authservice.repository.UserRepository;
import com.ecommerce.authservice.service.JwtService;
import com.ecommerce.authservice.service.UserService;
import java.time.Instant;
import java.util.Set;
import lombok.RequiredArgsConstructor;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

/**
 * @RequiredArgsConstructor remplace le constructeur explicite à 3 paramètres.
 */
@Service
@RequiredArgsConstructor
public class UserServiceImpl implements UserService {

    private final UserRepository userRepository;
    private final PasswordEncoder passwordEncoder;
    private final JwtService jwtService;

    @Override
    @Transactional
    public AuthResponse register(RegisterRequest request) {
        if (userRepository.existsByEmail(request.email())) {
            throw new IllegalArgumentException("Email déjà utilisé");
        }
        User user = new User();
        user.setEmail(request.email());
        user.setPasswordHash(passwordEncoder.encode(request.password()));
        user.setRoles(Set.of(Role.USER));
        userRepository.save(user);
        String token = jwtService.generateToken(user.getEmail(), Set.of("USER"));
        return new AuthResponse(token);
    }

    @Override
    @Transactional
    public AuthResponse login(AuthRequest request) {
        User user = userRepository.findByEmail(request.email())
                .orElseThrow(() -> new IllegalArgumentException("Identifiants invalides"));
        if (!passwordEncoder.matches(request.password(), user.getPasswordHash())) {
            throw new IllegalArgumentException("Identifiants invalides");
        }
        user.setLastLoginAt(Instant.now());
        String role = user.getRoles().contains(Role.ADMIN) ? "ADMIN" : "USER";
        String token = jwtService.generateToken(user.getEmail(), Set.of(role));
        return new AuthResponse(token);
    }
}

@RequestMapping("/api/auth")
@RequiredArgsConstructor

    }

import jakarta.persistence.*;
import java.math.BigDecimal;
import java.time.Instant;
import java.util.ArrayList;
import java.util.List;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * @Getter / @Setter remplacent tous les getters/setters manuels.
 * @NoArgsConstructor requis par JPA.
 * @Builder NON ajouté : items initialisée en ligne (new ArrayList<>()) et addItem()
 * maintient la cohérence bidirectionnelle — incompatible avec @Builder naïf.
 */
@Entity
@Table(name = "orders")
@Getter
@Setter
@NoArgsConstructor
public class Order {

    @Id
    @GeneratedValue(strategy = GenerationType.UUID)
    private String id;

    @Column(nullable = false)
    private String userId;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private OrderStatus status = OrderStatus.PENDING;

    @OneToMany(mappedBy = "order", cascade = CascadeType.ALL, orphanRemoval = true)
    private List<OrderItem> items = new ArrayList<>();

    @Column(nullable = false, precision = 10, scale = 2)
    private BigDecimal totalAmount;

    @Column(nullable = false)
    private Instant createdAt;

    @Version
    private Long version; // Optimistic locking

    /** Méthode métier conservée — maintient la cohérence bidirectionnelle JPA. */
    public void addItem(OrderItem item) {
        items.add(item);
        item.setOrder(this);
        recalculateTotal();
    }

    private void recalculateTotal() {
        this.totalAmount = items.stream()
            .map(OrderItem::getSubtotal)
            .reduce(BigDecimal.ZERO, BigDecimal::add);
    }
}

enum OrderStatus { PENDING, PAID, SHIPPED, DELIVERED, CANCELLED }

• @GeneratedValue(UUID) : Identifiants uniques sans coordination centralisée

• @Version : Verrouillage optimiste pour éviter les modifications concurrentes

• BigDecimal : Précision monétaire (jamais de double pour l'argent !)

    @ExceptionHandler(OrderNotFoundException.class)
    public ResponseEntity<Map<String, Object>> handleOrderNotFound(OrderNotFoundException ex) {
        return ResponseEntity.status(HttpStatus.NOT_FOUND)
                .body(Map.of(
                        "timestamp", Instant.now(),
                        "error", "NOT_FOUND",
                        "message", ex.getMessage()
                ));
    }

    @ExceptionHandler(MethodArgumentNotValidException.class)
    public ResponseEntity<Map<String, Object>> handleValidation(MethodArgumentNotValidException ex) {

                .map(fe -> Map.of(

                        "message", fe.getDefaultMessage()
                ))
                .toList();
        return ResponseEntity.badRequest()
                .body(Map.of(
                        "timestamp", Instant.now(),
                        "error", "VALIDATION_ERROR",
                        "details", errors
                ));
    }

        return ResponseEntity.badRequest()
                .body(Map.of(
                        "timestamp", Instant.now(),
                        "error", "BAD_REQUEST",

                ));
    }

package com.ecommerce.orderservice.service.impl;

import com.ecommerce.orderservice.domain.entity.Order;
import com.ecommerce.orderservice.domain.event.OrderCreatedEvent;
import com.ecommerce.orderservice.messaging.producer.OrderEventProducer;
import com.ecommerce.orderservice.repository.OrderRepository;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.time.Instant;

/**
 * @RequiredArgsConstructor remplace le constructeur explicite à 2 paramètres.
 * @Slf4j injecte : private static final Logger log = LoggerFactory.getLogger(OrderServiceImpl.class);
 */
@Slf4j
@Service
@RequiredArgsConstructor
public class OrderServiceImpl implements OrderService {

    private final OrderRepository orderRepository;
    private final OrderEventProducer eventProducer;

    @Override
    @Transactional
    public OrderResponse createOrder(CreateOrderRequest request, String userId) {
        // 1. Créer l'entité
        Order order = new Order();
        order.setUserId(userId);
        order.setCreatedAt(Instant.now());
        
        request.items().forEach(item -> 
            order.addItem(new OrderItem(item.productId(), item.quantity(), item.price()))
        );
        
        // 2. Persister en base
        Order saved = orderRepository.save(order);
        
        // 3. Publier l'événement
        eventProducer.sendOrderCreated(new OrderCreatedEvent(
            saved.getId(),
            saved.getUserId(),
            saved.getTotalAmount(),
            saved.getItems().stream().map(this::toItemEvent).toList()
        ));
        
        return toResponse(saved);
    }
}

Attention Production : L'événement est publié après la transaction. Si Kafka est indisponible, la commande existe en base mais l'événement n'est pas envoyé. Pour garantir la cohérence, utilisez le pattern Transactional Outbox (événements stockés en base puis publiés par un processus séparé).

package com.ecommerce.orderservice.messaging.producer;

import com.ecommerce.shared.event.OrderCreatedEvent;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.kafka.core.KafkaTemplate;
import org.springframework.kafka.support.SendResult;
import org.springframework.stereotype.Component;
import java.util.concurrent.CompletableFuture;

/**
 * @Slf4j génère : private static final Logger log = LoggerFactory.getLogger(OrderEventProducer.class);
 * @RequiredArgsConstructor génère le constructeur avec KafkaTemplate.
 */
@Slf4j
@Component
@RequiredArgsConstructor
public class OrderEventProducer {

    private static final String TOPIC = "orders";
    private final KafkaTemplate<String, Object> kafkaTemplate;

    public void sendOrderCreated(OrderCreatedEvent event) {
        // La clé détermine la partition (même clé = même partition = ordre garanti)
        String key = event.userId();
        
        CompletableFuture<SendResult<String, Object>> future = 
            kafkaTemplate.send(TOPIC, key, event);
        
        future.whenComplete((result, ex) -> {
            if (ex != null) {
                log.error("Échec envoi event {} : {}", event.eventId(), ex.getMessage());
            } else {
                log.info("Event {} envoyé sur partition {}, offset {}", 
                    event.eventId(),
                    result.getRecordMetadata().partition(),
                    result.getRecordMetadata().offset());
            }
        });
    }
}

Clé de partitionnement : En utilisant userId comme clé, toutes les commandes d'un même utilisateur vont dans la même partition, garantissant leur ordre de traitement.

export interface MfeEvent<T = unknown> {
  type: string;
  payload: T;
  source: string;
}

export class EventBus {
  static emit<T>(event: MfeEvent<T>): void {
    window.dispatchEvent(new CustomEvent('mfe-event', { detail: event }));
  }

  static on<T>(type: string, callback: (payload: T) => void): () => void {
    const handler = (e: CustomEvent<MfeEvent<T>>) => {
      if (e.detail.type === type) {
        callback(e.detail.payload);
      }
    };
    window.addEventListener('mfe-event', handler as EventListener);
    return () => window.removeEventListener('mfe-event', handler as EventListener);
  }
}

package com.ecommerce.paymentservice.service;

import com.ecommerce.shared.event.OrderCreatedEvent;
import com.ecommerce.shared.event.PaymentCompletedEvent;
import com.ecommerce.shared.event.PaymentFailedEvent;
import io.micrometer.observation.annotation.Observed;
import java.math.BigDecimal;
import java.time.Instant;
import java.util.UUID;
import lombok.extern.slf4j.Slf4j;
import org.springframework.stereotype.Service;

/**
 * @Slf4j remplace la déclaration manuelle du Logger.
 */
@Slf4j
@Service
public class PaymentService {

    @Observed(name = "payment.process")
    public void processPayment(OrderCreatedEvent event) {
        log.info("Traitement paiement pour orderId={}, amount={}",
                event.orderId(), event.totalAmount());

        // Pattern matching Java 21 sur le niveau de risque
        BigDecimal amount = event.totalAmount();
        String risk = switch (amount.compareTo(BigDecimal.valueOf(100))) {
            case -1 -> "LOW";
            case 0, 1 -> "HIGH";
            default -> "UNKNOWN";
        };
        log.info("Niveau de risque paiement: {}", risk);

        // Dans un vrai système, on appellerait un PSP externe.
        boolean success = true;

        if (success) {
            PaymentCompletedEvent completed = new PaymentCompletedEvent(
                    UUID.randomUUID().toString(),
                    event.orderId(),
                    UUID.randomUUID().toString(),
                    event.totalAmount(),
                    Instant.now()
            );
            log.info("Paiement complété: {}", completed);
        } else {
            PaymentFailedEvent failed = new PaymentFailedEvent(
                    UUID.randomUUID().toString(),
                    event.orderId(),
                    "Paiement refusé",
                    event.totalAmount(),
                    Instant.now()
            );
            log.warn("Paiement échoué: {}", failed);
        }
    }
}

Chapitre 14

Lombok — Intégration Non Destructive

Ajout d'annotations Lombok sur les classes Java existantes sans modifier le comportement fonctionnel. Approche strictement additive.

Pourquoi Lombok ?

Lombok est un processeur d'annotations qui génère du code boilerplate (getters, setters, constructeurs, builders) à la compilation. Il réduit significativement la verbosité des classes Java sans altérer la logique métier.

✂️
Moins de boilerplate
Suppression des getters/setters manuels — le code compile identiquement
🔒
Non-destructif
Aucune logique métier modifiée, aucune sérialisation cassée
🧩
Compatible Spring
@RequiredArgsConstructor remplace les constructeurs d'injection Spring
🚫
Jamais sur record
Les records Java 21 génèrent déjà accesseurs et equals/hashCode

Règle absolue : Lombok n'est JAMAIS ajouté sur les record (DTOs, événements). Les records Java 21 sont déjà immutables et génèrent automatiquement equals, hashCode, toString et des accesseurs. Ajouter Lombok serait redondant et pourrait créer des conflits.

Configuration Maven — Dépendance Lombok

Ajouter dans le pom.xml parent (ou dans chaque service) :

xmlpom.xml — dépendance Lombok
<!-- Lombok — scope provided : uniquement à la compilation -->
<dependency>
    <groupId>org.projectlombok</groupId>
    <artifactId>lombok</artifactId>
    <version>1.18.34</version>
    <scope>provided</scope>
</dependency>

<!-- Plugin maven-compiler-plugin : déclarer Lombok comme annotationProcessor -->
<plugin>
    <groupId>org.apache.maven.plugins</groupId>
    <artifactId>maven-compiler-plugin</artifactId>
    <configuration>
        <source>21</source>
        <target>21</target>
        <annotationProcessorPaths>
            <path>
                <groupId>org.projectlombok</groupId>
                <artifactId>lombok</artifactId>
                <version>1.18.34</version>
            </path>
        </annotationProcessorPaths>
    </configuration>
</plugin>

    private int quantity;

import jakarta.persistence.*;
import java.math.BigDecimal;
import java.time.Instant;
import java.util.ArrayList;
import java.util.List;
import lombok.Getter;
import lombok.NoArgsConstructor;
import lombok.Setter;

/**
 * Entité JPA Order.
 * @Builder NON ajouté : la collection items est initialisée en ligne (new ArrayList<>())
 * et addItem() maintient la cohérence bidirectionnelle — incompatible avec @Builder naïf.
 */
@Entity
@Table(name = "orders")
@Getter
@Setter
@NoArgsConstructor
public class Order {

    @Id
    @GeneratedValue(strategy = GenerationType.UUID)
    private String id;

    @Column(nullable = false)
    private String userId;

    @Enumerated(EnumType.STRING)
    @Column(nullable = false)
    private OrderStatus status = OrderStatus.PENDING;

    @OneToMany(mappedBy = "order", cascade = CascadeType.ALL, orphanRemoval = true)
    private List<OrderItem> items = new ArrayList<>();

    @Column(nullable = false, precision = 10, scale = 2)
    private BigDecimal totalAmount;

    @Column(nullable = false)
    private Instant createdAt;

    @Version
    private Long version;

    /** Méthode métier conservée — maintient la cohérence bidirectionnelle JPA. */
    public void addItem(OrderItem item) {
        items.add(item);
        item.setOrder(this);
        recalculateTotal();
    }

    private void recalculateTotal() {
        this.totalAmount = items.stream()
            .map(OrderItem::getSubtotal)
            .reduce(BigDecimal.ZERO, BigDecimal::add);
    }
}

import com.ecommerce.authservice.domain.dto.AuthRequest;
import com.ecommerce.authservice.domain.dto.AuthResponse;
import com.ecommerce.authservice.domain.dto.RegisterRequest;
import com.ecommerce.authservice.domain.entity.Role;
import com.ecommerce.authservice.domain.entity.User;
import com.ecommerce.authservice.repository.UserRepository;
import com.ecommerce.authservice.service.JwtService;
import com.ecommerce.authservice.service.UserService;
import java.time.Instant;
import java.util.Set;
import lombok.RequiredArgsConstructor;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;

@Service
@RequiredArgsConstructor
public class UserServiceImpl implements UserService {

    private final UserRepository userRepository;
    private final PasswordEncoder passwordEncoder;
    private final JwtService jwtService;

    @Override
    @Transactional
    public AuthResponse register(RegisterRequest request) {
        if (userRepository.existsByEmail(request.email())) {
            throw new IllegalArgumentException("Email déjà utilisé");
        }
        User user = new User();
        user.setEmail(request.email());
        user.setPasswordHash(passwordEncoder.encode(request.password()));
        user.setRoles(Set.of(Role.USER));
        userRepository.save(user);
        String token = jwtService.generateToken(user.getEmail(), Set.of("USER"));
        return new AuthResponse(token);
    }

    @Override
    @Transactional
    public AuthResponse login(AuthRequest request) {
        User user = userRepository.findByEmail(request.email())
                .orElseThrow(() -> new IllegalArgumentException("Identifiants invalides"));
        if (!passwordEncoder.matches(request.password(), user.getPasswordHash())) {
            throw new IllegalArgumentException("Identifiants invalides");
        }
        user.setLastLoginAt(Instant.now());
        String role = user.getRoles().contains(Role.ADMIN) ? "ADMIN" : "USER";
        String token = jwtService.generateToken(user.getEmail(), Set.of(role));
        return new AuthResponse(token);
    }
}

Dockerfile Optimisé — Multi-stage avec Cache Maven

dockerfilebackend/order-service/Dockerfile
# ─── ÉTAPE 1 : Téléchargement des dépendances Maven (cachée séparément) ───────
FROM eclipse-temurin:21-jdk-alpine AS deps
WORKDIR /app

# Copier uniquement les descripteurs Maven pour profiter du cache Docker
COPY pom.xml .
COPY .mvn/ .mvn/
COPY mvnw .
RUN chmod +x mvnw

# Télécharger les dépendances sans compiler (cache invalidé uniquement si pom.xml change)
RUN ./mvnw dependency:go-offline -B -q

# ─── ÉTAPE 2 : Compilation & tests ─────────────────────────────────────────
FROM deps AS builder
COPY src ./src

# Build avec tests (utilise le cache de dépendances de l'étape précédente)
RUN ./mvnw package -B -q

# Extraire les layers du JAR (optimisation démarrage Spring Boot 3.x)
RUN java -Djarmode=layertools -jar target/*.jar extract --destination target/extracted

# ─── ÉTAPE 3 : Image runtime minimale ──────────────────────────────────────
FROM eclipse-temurin:21-jre-alpine AS runtime

# Sécurité : utilisateur non-root dédié
RUN addgroup -S spring && adduser -S spring -G spring -u 1001
USER spring

WORKDIR /app

# Copier les layers dans l'ordre (de la moins volatile à la plus volatile)
# → Optimise le cache Docker pour les futures builds
COPY --from=builder --chown=spring:spring /app/target/extracted/dependencies/ ./
COPY --from=builder --chown=spring:spring /app/target/extracted/spring-boot-loader/ ./
COPY --from=builder --chown=spring:spring /app/target/extracted/snapshot-dependencies/ ./
COPY --from=builder --chown=spring:spring /app/target/extracted/application/ ./

# Health check Docker natif
HEALTHCHECK --interval=30s --timeout=5s --start-period=60s --retries=3 \
  CMD wget -q --spider http://localhost:8080/actuator/health || exit 1

EXPOSE 8080

# JVM tuning recommandé pour containers (Java 21 détecte automatiquement les CPU/RAM K8s)
ENV JAVA_OPTS="-XX:+UseContainerSupport -XX:MaxRAMPercentage=75.0 -XX:+UseG1GC -Djava.security.egd=file:/dev/./urandom"

ENTRYPOINT ["sh", "-c", "exec java $JAVA_OPTS org.springframework.boot.loader.launch.JarLauncher"]

name: CI/CD — Build, Test, Deploy DEV & TEST

on:
  push:
    branches:
      - main          # → déploiement DEV automatique
      - release/**    # → déploiement TEST automatique
  pull_request:
    branches: [main, release/**]

env:
  PROJECT_ID: ${{ secrets.GCP_PROJECT_ID }}
  REGISTRY: europe-west1-docker.pkg.dev
  REPO: ecommerce
  GKE_REGION: europe-west1-b

jobs:
  # ─── JOB 1 : TESTS ─────────────────────────────────────────────────────────
  test:
    name: Tests & Qualité
    runs-on: ubuntu-latest
    strategy:
      matrix:
        service: [auth-service, order-service, payment-service, inventory-service]
    steps:
      - uses: actions/checkout@v4

      - name: Set up JDK 21
        uses: actions/setup-java@v4
        with:
          java-version: '21'
          distribution: 'temurin'
          cache: maven

      - name: Run tests avec coverage
        run: |
          cd backend/${{ matrix.service }}
          mvn verify -B -q \
            -Djacoco.minimum.coverage=0.70 \
            -Dmaven.test.failure.ignore=false

      - name: Upload coverage report
        uses: codecov/codecov-action@v4
        with:
          token: ${{ secrets.CODECOV_TOKEN }}
          files: backend/${{ matrix.service }}/target/site/jacoco/jacoco.xml
          flags: ${{ matrix.service }}

  # ─── JOB 2 : BUILD & PUSH IMAGE ────────────────────────────────────────────
  build-and-push:
    name: Build Docker & Push Artifact Registry
    needs: test
    runs-on: ubuntu-latest
    if: github.event_name == 'push'
    outputs:
      image-tag: ${{ steps.meta.outputs.tag }}
    strategy:
      matrix:
        service: [auth-service, order-service, payment-service, inventory-service]
    steps:
      - uses: actions/checkout@v4

      - name: Auth GCP
        uses: google-github-actions/auth@v2
        with:
          credentials_json: ${{ secrets.GCP_SA_KEY }}

      - name: Configure Docker for Artifact Registry
        run: gcloud auth configure-docker ${{ env.REGISTRY }} --quiet

      - name: Set image metadata
        id: meta
        run: |
          SHORT_SHA=$(echo ${{ github.sha }} | cut -c1-8)
          BRANCH=$(echo ${{ github.ref_name }} | tr '/' '-')
          echo "tag=${SHORT_SHA}" >> $GITHUB_OUTPUT
          echo "full-image=${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPO }}/${{ matrix.service }}" >> $GITHUB_OUTPUT

      - name: Build Docker image (avec cache)
        uses: docker/build-push-action@v5
        with:
          context: ./backend/${{ matrix.service }}
          push: true
          tags: |
            ${{ steps.meta.outputs.full-image }}:${{ steps.meta.outputs.tag }}
            ${{ steps.meta.outputs.full-image }}:${{ github.ref_name == 'main' && 'dev-latest' || 'test-latest' }}
          cache-from: type=registry,ref=${{ steps.meta.outputs.full-image }}:buildcache
          cache-to: type=registry,ref=${{ steps.meta.outputs.full-image }}:buildcache,mode=max

      - name: Scan image (Trivy)
        uses: aquasecurity/trivy-action@master
        with:
          image-ref: "${{ steps.meta.outputs.full-image }}:${{ steps.meta.outputs.tag }}"
          format: 'sarif'
          output: 'trivy-results.sarif'
          severity: 'CRITICAL,HIGH'
          exit-code: '1'   # Bloque si CRITICAL détecté

  # ─── JOB 3 : DEPLOY DEV (branch main) ────────────────────────────────────
  deploy-dev:
    name: Deploy → DEV
    needs: build-and-push
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    environment:
      name: dev
      url: https://dev.api.ecommerce.internal
    steps:
      - uses: actions/checkout@v4

      - name: Auth GCP
        uses: google-github-actions/auth@v2
        with:
          credentials_json: ${{ secrets.GCP_SA_KEY }}

      - name: Get GKE credentials (DEV cluster)
        uses: google-github-actions/get-gke-credentials@v2
        with:
          cluster_name: ecommerce-dev
          location: ${{ env.GKE_REGION }}
          project_id: ${{ env.PROJECT_ID }}

      - name: Deploy via Kustomize (DEV)
        run: |
          SHORT_SHA=$(echo ${{ github.sha }} | cut -c1-8)
          cd infrastructure/kubernetes/overlays/dev
          
          # Mettre à jour les tags d'images
          for svc in auth-service order-service payment-service inventory-service; do
            kustomize edit set image \
              ${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPO }}/${svc}=${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPO }}/${svc}:${SHORT_SHA}
          done
          
          # Appliquer les manifestes
          kubectl apply -k .

      - name: Attendre rollout DEV
        run: |
          for svc in auth-service order-service payment-service inventory-service; do
            kubectl rollout status deployment/${svc} \
              -n ecommerce-dev \
              --timeout=300s
          done

      - name: Smoke tests DEV
        run: |
          sleep 10
          curl -f https://dev.api.ecommerce.internal/actuator/health || exit 1

      - name: Rollback si smoke test échoue
        if: failure()
        run: |
          for svc in auth-service order-service payment-service inventory-service; do
            kubectl rollout undo deployment/${svc} -n ecommerce-dev
          done

  # ─── JOB 4 : DEPLOY TEST (branch release/**) ──────────────────────────────
  deploy-test:
    name: Deploy → TEST
    needs: build-and-push
    runs-on: ubuntu-latest
    if: startsWith(github.ref, 'refs/heads/release/')
    environment:
      name: test
      url: https://test.api.ecommerce.com
    steps:
      - uses: actions/checkout@v4

      - name: Auth GCP
        uses: google-github-actions/auth@v2
        with:
          credentials_json: ${{ secrets.GCP_SA_KEY }}

      - name: Get GKE credentials (TEST cluster)
        uses: google-github-actions/get-gke-credentials@v2
        with:
          cluster_name: ecommerce-test
          location: ${{ env.GKE_REGION }}
          project_id: ${{ env.PROJECT_ID }}

      - name: Deploy via Kustomize (TEST)
        run: |
          SHORT_SHA=$(echo ${{ github.sha }} | cut -c1-8)
          cd infrastructure/kubernetes/overlays/test
          
          for svc in auth-service order-service payment-service inventory-service; do
            kustomize edit set image \
              ${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPO }}/${svc}=${{ env.REGISTRY }}/${{ env.PROJECT_ID }}/${{ env.REPO }}/${svc}:${SHORT_SHA}
          done
          
          kubectl apply -k .

      - name: Attendre rollout TEST
        run: |
          for svc in auth-service order-service payment-service inventory-service; do
            kubectl rollout status deployment/${svc} \
              -n ecommerce-test \
              --timeout=300s
          done

      - name: Integration tests TEST
        run: |
          sleep 15
          # Lancer les tests d'intégration complets contre l'env TEST
          mvn verify -pl integration-tests \
            -Dbase.url=https://test.api.ecommerce.com \
            -B -q

      - name: Rollback si tests d'intégration échouent
        if: failure()
        run: |
          for svc in auth-service order-service payment-service inventory-service; do
            kubectl rollout undo deployment/${svc} -n ecommerce-test
          done

      - name: Notifier sur Slack
        if: always()
        uses: 8398a7/action-slack@v3
        with:
          status: ${{ job.status }}
          text: "Déploiement TEST : ${{ job.status }} — SHA: ${{ github.sha }}"
          webhook_url: ${{ secrets.SLACK_WEBHOOK }}
